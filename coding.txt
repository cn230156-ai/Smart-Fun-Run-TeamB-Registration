// 1. When Button1 is clicked
document.getElementById("Button1").addEventListener("click", function() {
    BarcodeScanner1.DoScan();
});

// 2. After the BarcodeScanner gets the result (AfterScan)
function onAfterScan(result) {
    // Join the base URL with the scanned result
    let baseUrl = "https://script.google.com/macros/s/AKfycbzUTSq5l...";
    let url = baseUrl + result;
    
    // Send an HTTP GET Request (using fetch API to mimic Web1.Get)
    fetch(url)
        .then(response => response.text())
        .then(responseContent => {
            // Trigger the GotText logic upon a successful response
            onGotText(url, 200, "text/plain", responseContent);
        })
        .catch(error => console.error("Error:", error));
}

// 3. When Web1 receives the text response (GotText)
function onGotText(url, responseCode, responseType, responseContent) {
    // Get the current date and format it to "MMM d, yyyy" (e.g., Oct 24, 2026)
    let options = { month: 'short', day: 'numeric', year: 'numeric' };
    let currentDate = new Date().toLocaleDateString('en-US', options);
    
    // Join: responseContent + " Masa " + formatted date
    let finalString = responseContent + " Masa " + currentDate;
    
    // Display the result on Label1
    document.getElementById("Label1").innerText = finalString;
}
